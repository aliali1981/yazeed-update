    function populateStudents(className) {
        const studentSelect = document.getElementById('studentSelect');
        studentSelect.innerHTML = '<option value="" disabled selected>-- اختر اسم الطالب --</option>';
        if (studentsData[className]) {
            studentsData[className].forEach(name => {
                let opt = document.createElement('option');
                opt.value = name; opt.textContent = name;
                studentSelect.appendChild(opt);
            });
            document.getElementById('studentGroup').classList.remove('hidden');
            document.getElementById('dataFields').classList.remove('hidden');
        }
    }

    document.getElementById('updateForm').addEventListener('submit', async (e) => {
        e.preventDefault();
        const btn = document.getElementById('submitBtn');
        btn.disabled = true; btn.textContent = "جاري الإرسال...";

        try {
            await db.collection("phoneUpdates").add({
                class: document.getElementById('classSelect').value,
                student: document.getElementById('studentSelect').value,
                phone: document.getElementById('newPhone').value,
                relation: document.getElementById('relation').value,
                timestamp: firebase.firestore.FieldValue.serverTimestamp(),
                displayDate: new Date().toLocaleString('ar-OM')
            });
            alert("تم إرسال التحديث بنجاح");
            location.reload();
        } catch (e) { alert("خطأ في الإرسال"); btn.disabled = false; }
    });

    function openAdmin() {
        if(prompt("كلمة المرور:") === "Yazeed2026") {
            document.getElementById('userView').classList.add('hidden');
            document.getElementById('adminView').classList.remove('hidden');
            db.collection("phoneUpdates").orderBy("timestamp", "desc").onSnapshot(snap => {
                const tbody = document.getElementById('adminTableBody');
                tbody.innerHTML = "";
                snap.forEach(doc => {
                    const d = doc.data();
                    tbody.innerHTML += `<tr><td>${d.displayDate}</td><td>${d.class}</td><td>${d.student}</td><td>${d.phone}</td><td>${d.relation}</td></tr>`;
                });
            });
        }
    }
    function closeAdmin() { location.reload(); }
